class C( Int!? x ) {
	Void test() {
		x++; // Obviously x data must be constantiatized -- this is handled by rule "changing ctime value from rtime scope"

	}

	@ctime Int! a = 0;
	C( a ) c;

	c.test();

	c.test(); // What now? This is kinda screwed... a should be constantiatized or sumthing
